Как подключить: 

import LeafletMap from '../../components/Map/index.vue';

components: { LeafletMap }

Входные параметры: 

borderRadius:
type: string, value: '20px'

id:
type: string, value: 'empty' or 'your-value'

coordinates:
type: [Array]
value: [59.704574, 30.786810]

zoom:
type: Number, value: 20
